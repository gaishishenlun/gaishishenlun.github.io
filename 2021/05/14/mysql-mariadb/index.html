mysql/mariadb

Mysql/Mariadb 简单操作

常用

update user set password=password('123456') where user='root';	# 修改修改密码
grant all privileges on *.* to '用户名'@'%' identified by '密码' with grant option;	# 高权限用户，允许所有IP连接，用户可以不存在
flush privileges;	# 刷新数据库
use dbname;			# 打开数据库(dbname)
show databases;		# 显示所有数据库
show tables;		# 显示所有表
desc T_name;		# 显示表(T_name)的字段信息	
create database dbname);		# 创建名为dbname的数据库

数据库与数据表操作

1、导入导出

$ mysqldump -uroot -p密码 数据库名 > D:/备份文件名.sql		# 导出数据库
$ mysql -uroot -p密码 数据库名 < D:/备份文件名.sql			# 导入数据库

2、查看选择

show databases; 			# 查看所有数据库
show tables;				# 查看所有数据表
show create table T_name;	# 查看表的创建细节
desc T_name;				# 查看表结构
use dbname;					# 进入数据库
select database();  		# 查看当前所在的库

3、创建数据库

# CREATE DATABASE 数据库名
CREATE DATABASE Python_7; 
CREATE DATABASE pyrhon_7 charset utf8

# 修改数据库编码
alter database db1 charset gbk;

4、删除数据库

# drop database 需要删除的数据库名
drop database dbname;

5、新建表

# create table 新建数据表名（
#   	字段名 类型(长度) [约束],
#   	字段名 类型(长度) [约束]
#  );
create table class(
		id INT AUTO_INCREMENT,
	name varchar(32) UNIQUE,
	age varchar(32) NOT NULL
);

# 若根据已有的表创建新表
create table 新表 like 旧表  # 使用旧表创建新表
create table 新表 as select 字段1 字段2... from definition only	# 使用自定义值去新建表

6、查看表

select database();  		# 查看当前所在的库
show tables; 				# 查看数据库中所有表
desc T_name;				# 查看表结构
show create table T_name;	# 查看表的创建细节

7、删除表

#drop table 数据库表名
drop table T_name;

8、修改表

alter table 表名 add 字段名 类型（长度） [约束]; -- 添加列
alter table 表名 modify 字段名 类型（长度） [约束]; -- 修改列的类型长度及约束
alter table 表名 change 旧字段名 新字段名 类型（长度） [约束]; -- 修改列表名
alter table 表名 drop 字段名; -- 删除列
alter table 表名 character set 字符集; -- 修改表的字符集
rename table 表名 to 新表名; -- 修改表名

增删改查

1、插入数据

insert into T_name(字段名1，字段名2...) values(值1，值2...);	# 向表中插入某些列
insert into T_name values(值1，值2，值3， 值4...);			  # 向表中插入所有列

2、删除数据

delete from T_name where 条件;	# 删除符合where条件的数据，不加where则为所有数据
truncate table T_name;		# 重新建表，数据不能找回。执行速度比delete快

3、修改数据

update 表名 set 字段名=值,字段名=值...;		# 这个会修改所有的数据，把一列的值都变了
update 表名 set 字段名=值,字段名=值... where 条件;	# 只改符合where条件的行

4、查找数据

# 简单的数据查询
select * from T_name where 条件;		  # 查询满足where的所有列，不加where则为所有列
select 字段1,字段名2... from T_name;		# 查询指定列
